\subsection{Problem Solving}

\vspace{.25in}

\begin{mdframed}
\textbf{Strategy}
\begin{enumerate}
\item Read and re-read the problem.
\item Determine what is given and what is needed/unknown.
\item Use a variable to represent the unknown quantity.
\item Write an equation using the variable to model the scenario.
\item Solve for the unknown.
\end{enumerate}
\end{mdframed}

\vspace{.15in}

\begin{example}
Four subtracted from six times a number is sixty-eight. What is the number?
\vspace{1in}
\end{example}

\begin{example}
Page numbers on facing pages of a book are \emph{consecutive integers}. Two pages that face each other have a sum of 145. What are the page numbers?
\end{example}

\newpage

\begin{example}
A taxi charges \$2.00 to start the meter and \$0.25 for each mile driven. How far can you travel if you have \$10.00?
\vspace{2.75in}
\end{example}

\begin{example}
A rectangular pool is three times as long as it is wide. If the perimeter is 320 feet, what are the dimensions of the pool?
\end{example}

\newpage

\begin{example}
After a 40\% price reduction, an exercise machine sold for \$564. What was the original price?
\end{example}
